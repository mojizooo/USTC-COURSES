### 5.

 ``` verilog
 module sum_adder(
 	input Load,
 	input clk,
 	input rst,
     input [15:0] load,
     input In,
     output [15:0] out);
     reg [15:0] result;
     always @ (posedge clk or negedge rst)
         begin
             if(!rst)
                 result <= 0;
             else if(Load)
                 result <= load;
             else
                 begin
                     result <= result + In;
                 end
         end
     always @ (posedge clk)
         begin
             out <= result;
         end
 endmodule        
 ```
