### 7.27

```c
int order = 0;
int visit[MAX];
int Path(ALGraph g, int v, int w, int k) {
	ArcNode *p;
	
	visit[v] = ++order; //visit数组用于标记目前是第几个
	if (v == w && k == 0) return 1; //找到目标路径
	else if (k > 0) {
		//进行深度优先搜索
		for (p = g.vexs[v].firstarc; p; p = p->nextarc) {
			if (visit[p->adjV] == 0) { //没有经过
				if (Path(g, p->adjv, w, k - 1))
					return 1;
				else
					visit[p->adjv] = 0;
                order --; //此路不通换路
			}
		}
	}
	return 0;
}
```
